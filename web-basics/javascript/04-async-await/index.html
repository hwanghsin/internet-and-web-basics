非同步：Async / Await

來源：js_raw.txt｜從 Callback → Promise → async/await 的演進，以及 Promise 組合方法

演進史

// ❶ Callback（回調地獄 Callback Hell）
fetchUser(userId, function(user) {
  fetchOrders(user.id, function(orders) {
    fetchProduct(orders[0].productId, function(product) {
      console.log(product);           // 三層巢狀，越來越難維護
    }, handleError);
  }, handleError);
}, handleError);


// ❷ Promise 鏈（.then() 扁平化）
fetchUser(userId)
  .then(user    => fetchOrders(user.id))
  .then(orders  => fetchProduct(orders[0].productId))
  .then(product => console.log(product))
  .catch(err    => console.error(err));  // 統一錯誤處理


// ❸ async/await（最易讀，語法糖包裝 Promise）
async function loadProduct(userId) {
  try {
    const user    = await fetchUser(userId);
    const orders  = await fetchOrders(user.id);
    const product = await fetchProduct(orders[0].productId);
    console.log(product);
  } catch (err) {
    console.error('流程中斷：', err.message);
  } finally {
    console.log('無論成功失敗都執行');
  }
}

async/await 執行順序

await 會暫停目前 async function，讓主執行緒繼續跑其他同步程式碼

console.log('1. 同步開始');

async function run() {
  console.log('2. async function 開始');
  const result = await delay(500);   // 暫停 async，讓出主執行緒
  console.log('4. await 結束，result =', result);
}

run();  // 開始執行但遇到 await 就暫停
console.log('3. 同步繼續（async 暫停中）');
// 輸出順序：1 → 2 → 3 → 4

並行（Parallel）vs 循序（Sequential）

多個非同步操作時，選對方式可以大幅縮短等待時間

// ❌ 循序（total = A + B + C 的時間）
const a = await fetchA();   // 等 1s
const b = await fetchB();   // 再等 1s
const c = await fetchC();   // 再等 1s  → 共 3s

// ✅ Promise.all 並行（total = max(A, B, C) 的時間）
const [a, b, c] = await Promise.all([
  fetchA(),   // 同時開始
  fetchB(),   // 同時開始
  fetchC(),   // 同時開始
]);  // → 共約 1s（最慢那個）

Promise 靜態方法比較

方法何時 resolve何時 reject用途
Promise.all 全部成功 任一失敗就立刻 reject 全部都需要才能繼續（如同時載入必要資源）
Promise.allSettled 全部完成（不管成敗） 永不 reject 需要知道每個結果（如批次操作）
Promise.race 第一個完成（不管成敗） 第一個失敗 超時控制（timeout race）
Promise.any 第一個成功 全部都失敗 選最快的成功（多個備援 API）

錯誤處理：try / catch / finally

async function safeFetch(url) {
  try {
    const response = await fetch(url);

    // ① fetch 的特性：HTTP 4xx/5xx 不會 throw
    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`);
    }

    // ② JSON 解析失敗也要捕獲
    const data = await response.json();
    return data;

  } catch (err) {
    if (err instanceof TypeError) {
      console.error('網路錯誤（DNS/TCP 失敗）：', err.message);
    } else {
      console.error('其他錯誤：', err.message);
    }
    throw err;  // 再次 throw 讓上層決定如何處理

  } finally {
    console.log('載入完成（無論成敗）');
    hideLoadingSpinner();
  }
}

Generator / yield（進階）

Generator 是 async/await 的底層機制，yield 讓函式可以「暫停」和「恢復」

// Generator 函式（function* 有星號）
function* counter() {
  yield 1;   // 暫停，回傳 {value:1, done:false}
  yield 2;   // 暫停，回傳 {value:2, done:false}
  yield 3;   // 暫停，回傳 {value:3, done:false}
  return 'done';  // 結束，回傳 {value:'done', done:true}
}

const gen = counter();
gen.next();  // {value:1, done:false}
gen.next();  // {value:2, done:false}
gen.next();  // {value:3, done:false}
gen.next();  // {value:'done', done:true}

// async/await 的底層就是 Generator + Promise 的語法糖
