const li = document.createElement('li');
li.textContent = '新項目';
li.style.padding = '6px';
list.appendChild(li);                    // 加到最後

// 插入到特定位置
list.insertBefore(li, list.firstChild);  // 插到第一個

// 替換節點
list.replaceChild(newNode, oldNode);

// 刪除節點
list.removeChild(list.lastChild);

// 修改屬性

classList 操作

這個 div 的 class 可以被動態修改 
el.classList.add('active')       // 加上 class
el.classList.remove('active')    // 移除 class
el.classList.toggle('active')    // 有就移除，沒有就加上
el.classList.contains('active')  // 檢查是否包含（回傳 boolean）
el.classList.replace('old','new')// 替換 class 名稱

// 屬性操作
el.getAttribute('data-status')        // 讀取屬性
el.setAttribute('data-status','done') // 設定屬性
el.removeAttribute('disabled')        // 移除屬性
el.hasAttribute('hidden')             // 檢查是否有屬性

// dataset（data-* 屬性的快捷）
el.dataset.status  // 等於 getAttribute('data-status')

// 直接改 style
el.style.color = '#68d391';
el.style.fontSize = '1.2em';
